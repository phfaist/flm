% Define a custom type for flm with empty type names,
% so \zcref just prints the raw ref text with no prefix like "Theorem" or "Equation".
\zcRefTypeSetup{flm--internal--ref}{
  Name-sg = ,       % no name prefix (singular)
  name-sg = ,       % no name prefix (singular, lowercase)
  Name-pl = ,       % no name prefix (plural)
  name-pl = ,       % no name prefix (plural, lowercase)
}
\newcounter{flm@internal@refcnt}

\newcommand{\flmDefLabelText}[2]{%
  \begingroup
    % zref will get the formatted counter value, so hack that to return our
    % full display text for this label
    \def\theflm@internal@refcnt{#1}%
    % Step the counter so zref sees a valid counter context
    \refstepcounter{flm@internal@refcnt}%
    % Override the type to our flm internal type
    \zcsetup{reftype=flm--internal--ref}%
    #2%
  \endgroup
}
\makeatother

\makeatletter
\ExplSyntaxOn
\zref@newprop{refwithpage}{%
  \str_case:onF { \@currentcounter }
    {
      {equation} { (\@currentlabel) }
      % {section}  { \@currentlabel }
      % {subsection} { \@currentlabel }
    }
    { \@currentlabel } % fallback
  \raisebox{2pt}{\tiny\(\rightarrow\)p.\,\thepage}%
}
\zref@addprop{main}{refwithpage}
\NewDocumentCommand{\myzcref}{m}{%
  \zcref{#1}%
}
\zcsetup{
  nameinlink,
  cap,
  ref=refwithpage,
  refbounds={,,,},
}
\ExplSyntaxOff
\makeatother
